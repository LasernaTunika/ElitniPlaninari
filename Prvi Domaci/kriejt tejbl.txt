CREATE TABLE AktivistaStranke
(
ID INTEGER NOT NULL PRIMARY KEY,
Lime VARCHAR2(63),
ImeRod VARCHAR2(63),
Prezime VARCHAR2(63),
Ulica VARCHAR2(63),
Broj INTEGER,
Grad VARCHAR2(63),
IDKoord INTEGER FOREIGN KEY REFERENCES Koordinator (IDAkt),
IDGM INTEGER FOREIGN KEY REFERENCES GlasackoMesto (ID)
);

CREATE TABLE Koordinator
(
IDAkt INTEGER NOT NULL PRIMARY KEY REFERENCES AktivistaStranke (ID),
Opstina VARCHAR2(63),
Ulica VARCHAR2(63),
Broj INTEGER,
Grad VARCHAR2(63)
);
